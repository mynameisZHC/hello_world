class Solution {
public:
    int fib(int n) {
        if (n == 0 || n == 1) {
            return n;
        }
        int num1 = 0;
        int num2 = 1;
        int index = 2;
        int num;
        while (index <= n) {
            num = num1 + num2;
            num1 = num2;
            num2 = num;
            index++;
        }
        return num;
    }
};
