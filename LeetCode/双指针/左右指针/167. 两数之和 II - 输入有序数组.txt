class Solution {
public:
    vector<int> twoSum(vector<int>& numbers, int target) {
        int left = 0;
        int right = numbers.size() - 1;
        while (left < right) {
            int sumNumber = numbers[left] + numbers[right];
            if (sumNumber == target) {
                return {left + 1, right + 1};
            } else if (sumNumber < target) {
                left++;
            } else if (sumNumber > target) {
                right--;
            }
        }
        return {-1, -1};
    }
};
