class Solution {
public:
    TreeNode* dfs(const vector<int> &nums, int leftIndex, int rightIndex)
    {
        if (leftIndex >= rightIndex) {
            return nullptr;
        }
        int maxNumIndex = leftIndex;
        for (int index = leftIndex; index < rightIndex; index++) {
            if (nums[maxNumIndex] < nums[index]) {
                maxNumIndex = index;
            }
        }
        TreeNode *node = new TreeNode(nums[maxNumIndex]);
        node->left = dfs(nums, leftIndex, maxNumIndex);
        node->right = dfs(nums, maxNumIndex + 1, rightIndex);
        return node;
    }

    TreeNode* constructMaximumBinaryTree(vector<int>& nums) {
        return dfs(nums, 0, nums.size());
    }
};
